CREATE DATABASE BD_suplementos;

USE BD_suplementos;

CREATE TABLE TB_CLIENTE
(
ID_CLIENTE INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
NOME VARCHAR(40) NOT NULL,
ENDERECO VARCHAR(60) NOT NULL,
CIDADE VARCHAR(40) NOT NULL,
CPF CHAR(11) NOT NULL
)ENGINE=INNODB;
SELECT*FROM tb_cliente;

CREATE TABLE TB_funcionario
(
id_funcionario INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
NOME VARCHAR(40) NOT NULL,
ENDERECO VARCHAR(60) NOT NULL,
CIDADE VARCHAR(40) NOT NULL,
CPF CHAR(11) NOT NULL
)ENGINE=INNODB;
SELECT*FROM tb_funcionario;

CREATE TABLE TB_produto
(
id_produto INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
NOME_produto VARCHAR(40) NOT NULL,
preco_unitario VARCHAR(60) NOT NULL,
idcategoria VARCHAR(40) 

)ENGINE=INNODB;
SELECT*FROM tb_produto;

CREATE TABLE TB_categoria
(
id_categoria INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
NOME_categoria VARCHAR(40) NOT NULL

)ENGINE=INNODB;
SELECT*FROM tb_categoria;

CREATE TABLE TB_pedido
(

id_pedido INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
valor_final VARCHAR(60) NOT NULL,
hora VARCHAR(60) NOT NULL,
quantidade VARCHAR(60) NOT NULL,
data VARCHAR(40) NOT NULL,
idcliente VARCHAR(60),
idproduto VARCHAR(60),
idfuncionario VARCHAR(40)

)ENGINE=INNODB;
SELECT*FROM tb_pedido;
